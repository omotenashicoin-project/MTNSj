public class ECHO256 extends ECHOSmallCore {

	/**
	 * Create the engine.
	 */
	public ECHO256()
	{
		super();
	}

	/** @see Digest */
	public int getDigestLength()
	{
		return 32;
	}

	/** @see Digest */
	public Digest copy()
	{
		return copyState(new ECHO256());
	}
}
