package com.hashengineering.utils;

/**
 * Created by HashEngineering on 7/2/14.
 */
public class ByteArrayUtils {

    public static byte[] trim256(byte [] bytes)
    {
        byte [] result = new byte[32];
        for (int i = 0; i < 32; i++){
            result[i] = bytes[i];
        }
        return result;
    }
}
